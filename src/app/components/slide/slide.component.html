How to import 
// inside your app.component.ts

import {{ '{ MPortalService } ' }} from '@mamouth/ui';


public openSlide(): void {{'{'}}
    const portal: MPortalService<{{'SlideTestComponent'}}> = new MPortalService<{{'SlideTestComponent'}}>(); 
    const slide: SlideTestComponent = portal.open(SlideTestComponent)
    slide.close.subscribe(() => portal.close());
{{'}'}}


Notice: {{'SlideTestComponent'}} is the component that you want to display in Portal
